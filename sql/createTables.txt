Address Table

CREATE TABLE `test`.`address` (
  `ID` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  `ApartmentNo` VARCHAR(45) NOT NULL,
  `Street` VARCHAR(45) NOT NULL,
  `City` VARCHAR(45) NOT NULL,
  `State` VARCHAR(45) NOT NULL,
  `Country` VARCHAR(45) NOT NULL,
  `Zip` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`ID`)
)
ENGINE = InnoDB;

User table – 
CREATE TABLE `test`.`user` (
  `ID` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  `Name` VARCHAR(45) NOT NULL,
  `Phone` VARCHAR(45) NOT NULL,
  `DOB` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`ID`),
  CONSTRAINT `AddressID` FOREIGN KEY `AddressID` (`ID`)
    REFERENCES `address` (`ID`)
    ON DELETE RESTRICT
    ON UPDATE RESTRICT
)
ENGINE = InnoDB;


Vehicle table

CREATE TABLE `test`.`vehicle` (
  `ID` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  `Name` VARCHAR(45) NOT NULL,
  `Classs` VARCHAR(45) NOT NULL,
  `Model` VARCHAR(45) NOT NULL,
  `Make` VARCHAR(45) NOT NULL,
  `Color` VARCHAR(45) NOT NULL,
  `Odometer` VARCHAR(45) NOT NULL,
  `Transmission` VARCHAR(45) NOT NULL,
  `Fuel` VARCHAR(45) NOT NULL,
  `Drive` VARCHAR(45) NOT NULL,
  `Type` VARCHAR(45) NOT NULL,
  `WIN` VARCHAR(45) NOT NULL,
  `NumberPlate` VARCHAR(45) NOT NULL,
  `Status` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`ID`)
)
ENGINE = InnoDB;


Rent table

CREATE TABLE `test`.`rent` (
  `ID` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  `Pickdate` DATETIME NOT NULL,
  `RetrunDate` DATETIME NOT NULL,
  `TotalRent` VARCHAR(45) NOT NULL,
  `GPS` BOOLEAN NOT NULL,
  `Tax` VARCHAR(45) NOT NULL,
  `ConfirmationNo` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`ID`),
  CONSTRAINT `UserID` FOREIGN KEY `UserID` (`ID`)
    REFERENCES `user` (`ID`)
    ON DELETE RESTRICT
    ON UPDATE RESTRICT,
  CONSTRAINT `PickupLocation` FOREIGN KEY `PickupLocation` (`ID`)
    REFERENCES `address` (`ID`)
    ON DELETE RESTRICT
    ON UPDATE RESTRICT,
  CONSTRAINT `ReturnLocation` FOREIGN KEY `ReturnLocation` (`ID`)
    REFERENCES `address` (`ID`)
    ON DELETE RESTRICT
    ON UPDATE RESTRICT,
  CONSTRAINT `VehicleID` FOREIGN KEY `VehicleID` (`ID`)
    REFERENCES `vehicle` (`ID`)
    ON DELETE RESTRICT
    ON UPDATE RESTRICT
)
ENGINE = InnoDB;
